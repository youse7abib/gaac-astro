function getAmbassadorData() {
  const projectId = 'gaac-ambassador';
  const apiKey = 'AIzaSyBZP5nwK9-C5HOJK6JEkMK4vylH9vkoKss';
  const url = `https://firestore.googleapis.com/v1/projects/${projectId}/databases/(default)/documents/ambassadors?key=${apiKey}`;

  const res = UrlFetchApp.fetch(url);
  const data = JSON.parse(res.getContentText());
  const sheet = SpreadsheetApp.getActiveSheet();

  sheet.getDataRange().clearContent();

  // Headers
  const headers = [
    'Name', 'Email', 'Phone', 'Country', 'School',
    'Referral Code', 'Points', 'Successful Registrations',
    'Registered At (Egypt Time)'
  ];
  sheet.appendRow(headers);

  // Collect rows first to sort by newest first
  const rows = [];

  (data.documents || []).forEach(doc => {
    const f = doc.fields || {};

    // Convert createdAt to Egypt time zone
    let egyptTime = '';
    if (f.createdAt?.stringValue) {
      const date = new Date(f.createdAt.stringValue);
      egyptTime = Utilities.formatDate(date, 'Africa/Cairo', 'yyyy-MM-dd HH:mm:ss');
    }

    rows.push([
      f.fullName?.stringValue || '',
      f.email?.stringValue || '',
      f.phone?.stringValue || '',
      f.country?.stringValue || '',
      f.school?.stringValue || '',
      f.referralCode?.stringValue || '',
      f.points?.integerValue || 0,
      f.successfulRegistrations?.integerValue || 0,
      egyptTime
    ]);
  });

  // Sort newest first (reverse chronological)
  rows.sort((a, b) => {
    if (a[8] > b[8]) return -1;
    if (a[8] < b[8]) return 1;
    return 0;
  });

  // Write all rows at once
  if (rows.length > 0) {
    sheet.getRange(2, 1, rows.length, headers.length).setValues(rows);
  }
}
